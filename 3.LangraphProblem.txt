3a. Network Issue Detector
Logic:

python
def detect_network_issues(state):
    issues = []
    
    # Check slice failures
    slice_failures = group_by_slice(state.raw_transactions)
    for slice_id, txns in slice_failures.items():
        failure_rate = sum(1 for t in txns if t['status'] == 'failure') / len(txns)
        if failure_rate > 0.3:  # 30% threshold
            issues.append({
                "type": "slice_failure",
                "slice_id": slice_id,
                "failure_rate": failure_rate,
                "affected_txns": len(txns)
            })
    
    # Check high latency
    high_latency = [t for t in state.raw_transactions if t['latency'] > 500]
    if len(high_latency) > 50:
        issues.append({
            "type": "high_latency",
            "count": len(high_latency),
            "avg_latency": mean([t['latency'] for t in high_latency])
        })
    
    return {"network_issues": issues}
Signals:

Network slice drops (failure_rate > threshold)
High latency (> 500ms)
Bandwidth degradation
Correlated failures by slice_id

Logic:

def detect_fraud_patterns(state):
    fraud_signals = []
    
    # Explicit fraud flags
    flagged = [t for t in state.raw_transactions if t['fraud_flag'] == True]
    
    # Velocity checks (same account, multiple txns)
    account_velocity = group_by_account(state.raw_transactions)
    for account, txns in account_velocity.items():
        if len(txns) > 10 and time_span(txns) < 3600:  # 10+ txns in 1 hour
            fraud_signals.append({
                "type": "velocity_anomaly",
                "account": account,
                "txn_count": len(txns),
                "total_amount": sum(t['amount'] for t in txns)
            })
    
    # Geolocation jumps
    geo_jumps = detect_impossible_travel(state.raw_transactions)
    
    # Large amount anomalies
    large_txns = [t for t in state.raw_transactions 
                  if t['amount'] > 10000 and t['device'] == 'new_device']
    
    return {
        "fraud_signals": fraud_signals,
        "flagged_transactions": flagged,
        "geo_anomalies": geo_jumps,
        "large_amount_txns": large_txns
    }
Signals:

Explicit fraud_flag = True
Transaction velocity anomalies
Geolocation impossibilities
Large amounts + new device
Unusual spending patterns

def analyze_spending_patterns(state):
    patterns = {
        "high_value_txns": [],
        "device_changes": [],
        "normal_activity": {}
    }
    
    # High-value transactions
    high_value = [t for t in state.raw_transactions if t['amount'] > 5000]
    patterns["high_value_txns"] = high_value
    
    # Device changes
    device_changes = detect_device_switches(state.raw_transactions)
    patterns["device_changes"] = device_changes
    
    # Normal baseline
    normal = [t for t in state.raw_transactions 
              if t['status'] == 'success' 
              and not t['fraud_flag']
              and t['amount'] < 5000]
    
    patterns["normal_activity"] = {
        "count": len(normal),
        "avg_amount": mean([t['amount'] for t in normal]),
        "transaction_types": Counter([t['type'] for t in normal])
    }
    
    return {"spending_patterns": patterns}
Signals:

Large purchases (> $5000)
New device usage
Normal transaction baseline
Spending trends

5. Classification & Routing Decision Node
Logic:

python
def route_decision(state):
    story = state['story_synthesis']
    
    # Rule-based overrides
    if len(state['network_issues']) > 0 and story['impact_level'] == 'high':
        department = "IT"
        priority = "P1"
    elif len(state['fraud_signals']) > 5:
        department = "Fraud"
        priority = "P1" if story['confidence'] > 0.8 else "P2"
    elif story['recommended_department'] == "General Support":
        department = "General Support"
        priority = "P3"
    else:
        department = story['recommended_department']
        priority = "P2"
    
    # Human-in-loop trigger
    needs_review = (
        story['confidence'] < 0.7 or
        story['impact_level'] == 'high' or
        len(state['fraud_signals']) > 10
    )
    
    return {
        "target_department": department,
        "priority": priority,
        "needs_human_review": needs_review
    }
Routing Logic:

IT: Network slice failures, high latency, infrastructure issues
Fraud: Fraud flags, velocity anomalies, geo jumps
General Support: Normal activity spikes, customer queries, trends

LangGraph Implementation Code
python
from langgraph.graph import StateGraph, END
from typing import TypedDict, List, Dict, Annotated
import operator
 
# State definition
class TransactionAnalysisState(TypedDict):
    raw_transactions: List[Dict]
    aggregated_metrics: Dict
    network_issues: Annotated[List[Dict], operator.add]
    fraud_signals: Annotated[List[Dict], operator.add]
    spending_patterns: Dict
    story_synthesis: Dict
    target_department: str
    priority: str
    needs_human_review: bool
    report: Dict
    approved: bool
 
# Node functions
def ingest_data(state):
    # Load transactions
    transactions = fetch_hourly_transactions()
    return {"raw_transactions": transactions, "transaction_count": len(transactions)}
 
def extract_features(state):
    metrics = compute_aggregated_metrics(state['raw_transactions'])
    return {"aggregated_metrics": metrics}
 
def detect_network_issues(state):
    issues = analyze_network_patterns(state['raw_transactions'])
    return {"network_issues": issues}
 
def detect_fraud(state):
    signals = analyze_fraud_patterns(state['raw_transactions'])
    return {"fraud_signals": signals}
 
def analyze_spending(state):
    patterns = analyze_spending_patterns(state['raw_transactions'])
    return {"spending_patterns": patterns}
 
def synthesize_story(state):
    # LLM call
    story = llm_generate_narrative(state)
    return {"story_synthesis": story}
 
def route_decision(state):
    decision = classify_and_route(state)
    return decision
 
def generate_report(state):
    dept = state['target_department']
    if dept == "IT":
        report = generate_it_report(state)
    elif dept == "Fraud":
        report = generate_fraud_report(state)
    else:
        report = generate_support_report(state)
    return {"report": report}
 
def deliver_report(state):
    send_to_department(state['target_department'], state['report'])
    return {"approved": True}
 
def human_review(state):
    # Wait for human input
    approval = request_human_approval(state['report'])
    return {"approved": approval}
 
# Conditional edges
def should_review(state):
    return "review" if state['needs_human_review'] else "deliver"
 
def after_review(state):
    return "deliver" if state['approved'] else "regenerate"
 
# Build graph
workflow = StateGraph(TransactionAnalysisState)
 
# Add nodes
workflow.add_node("ingest", ingest_data)
workflow.add_node("extract", extract_features)
workflow.add_node("detect_network", detect_network_issues)
workflow.add_node("detect_fraud", detect_fraud)
workflow.add_node("analyze_spending", analyze_spending)
workflow.add_node("synthesize", synthesize_story)
workflow.add_node("route", route_decision)
workflow.add_node("generate", generate_report)
workflow.add_node("review", human_review)
workflow.add_node("deliver", deliver_report)
 
# Add edges
workflow.set_entry_point("ingest")
workflow.add_edge("ingest", "extract")
 
# Parallel pattern detection
workflow.add_edge("extract", "detect_network")
workflow.add_edge("extract", "detect_fraud")
workflow.add_edge("extract", "analyze_spending")
 
# Converge to synthesis
workflow.add_edge("detect_network", "synthesize")
workflow.add_edge("detect_fraud", "synthesize")
workflow.add_edge("analyze_spending", "synthesize")
 
workflow.add_edge("synthesize", "route")
workflow.add_edge("route", "generate")
 
# Conditional human review
workflow.add_conditional_edges(
    "generate",
    should_review,
    {
        "review": "review",
        "deliver": "deliver"
    }
)
 
workflow.add_conditional_edges(
    "review",
    after_review,
    {
        "deliver": "deliver",
        "regenerate": "generate"
    }
)
 
workflow.add_edge("deliver", END)
 
# Compile
app = workflow.compile()
